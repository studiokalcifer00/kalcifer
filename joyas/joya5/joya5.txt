<script>
  // Mostrar el a√±o actual en el footer
  document.getElementById('year').textContent = new Date().getFullYear();

  // Carrusel de im√°genes
  let currentSlide = 0;
  const slides = document.querySelectorAll('.carousel img');

  function showSlide(index) {
    slides.forEach((s, i) => s.classList.toggle('active', i === index));
  }

  function prevSlide() {
    currentSlide = (currentSlide - 1 + slides.length) % slides.length;
    showSlide(currentSlide);
  }

  function nextSlide() {
    currentSlide = (currentSlide + 1) % slides.length;
    showSlide(currentSlide);
  }

  // Variables para guardar los colores seleccionados
  let selectedPurpurina = null;
  let selectedTinte = null;

  // Funci√≥n para seleccionar color
  function selectColor(name, element) {
    const group = element.closest('.color-group').querySelector('h4').textContent;
    const options = element.closest('.color-row').querySelectorAll('.color-option');
    options.forEach(opt => opt.classList.remove('selected'));
    element.classList.add('selected');

    if (group.includes("Purpurina")) {
      selectedPurpurina = name;
    } else if (group.includes("Tinte")) {
      selectedTinte = name;
    }
  }

  // Validaci√≥n y a√±adir al carrito
  function addToCartWithColor() {
    if (!selectedPurpurina || !selectedTinte) {
      alert("‚ö†Ô∏è Debes seleccionar un color de Purpurina y un Tinte antes de a√±adir al carrito.");
      return;
    }

    const item = {
      name: "COLLAR GATO",
      price: 10,
      color: `Purpurina: ${selectedPurpurina}, Tinte: ${selectedTinte}`
    };

    addToCart(item);
  }

  // Funciones del carrito
  function loadCart() {
    const cart = JSON.parse(localStorage.getItem('kalcifer_cart')) || [];
    document.getElementById('cart-count').textContent = cart.length;
    return cart;
  }

  function saveCart(cart) {
    localStorage.setItem('kalcifer_cart', JSON.stringify(cart));
    loadCart();
  }

  function addToCart(item) {
    let cart = loadCart();
    cart.push(item);
    saveCart(cart);
    alert("‚úÖ Producto a√±adido al carrito:\n" + item.color);
  }

  // Inicializar el carrito al cargar la p√°gina
  loadCart();
</script>
